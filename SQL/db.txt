
CREATE TABLE CLOTHING(
SEQ NUMBER(20) NOT NULL,    --상품게시글 번호
CANUM NUMBER(8)  NOT NULL,   --상품 카테고리 번호 (바지,긴판,반팔,아웃도어)
ITEMNUM NUMBER(8) PRIMARY KEY,  --상품 번호
TITLE VARCHAR2(100) NOT NULL,     --상품명
PRICE NUMBER(20) NOT NULL,     --상품 가격
SIZE VARCHAR2(20) NOT NULL,   --사이즈
BRAND VARCHAR2(30) NOT NULL,  --상품 브랜드
COUNT NUMBER(8) NOT NULL,      --상품 수량
CONTENT VARCHAR2(50) NOT NULL,   --게시글 내용
MAINPT VARCHAR2(50) NOT NULL,  --상품 썸네일 사진
NEWMAINPT VARCHAR2(50) NOT NULL, --상품 썸네일 사진
RVCOUNT NUMBER(20) NOT NULL,     -- 리뷰 갯수
RDATE DATE NOT NULL          --상품 등록일
);

CREATE SEQUENCE SEQ_CLOTHING    --시퀀스 값
START WITH 1 INCREMENT BY 1;

CREATE SEQUENCE ITEMNUM_CLOTHING    --아이템넘버 시퀀스 값
START WITH 100 INCREMENT BY 1;

-----

CREATE TABLE CART(
SEQ NUMBER(20) NOT NULL,    --장바구니 상품번호
ID VARCHAR2(50) NOT NULL,    -- 회원아이디
CHECKNUM NUMBER(2) NOT NULL,  --결제 확인용
CANUM NUMBER(2)  NOT NULL,   --상품 카테고리 번호 (바지,긴판,반팔,아웃도어)
ITEMNUM NUMBER(8) NOT NULL,  --상품 번호 (CLOTHING에 ITEMNUM 와 연결)
MAINPT VARCHAR2(50) NOT NULL,  --상품 썸네일 사진
TITLE VARCHAR2(50) NOT NULL,     --상품명
COUNT NUMBER(8) NOT NULL,      --상품 수량
PRICE NUMBER(20) NOT NULL,     --상품 가격
ADDRESS VARCHAR2(100) NOT NULL              --배송지
);

ALTER TABLE CART
ADD CONSTRAINT FK_CART_ITEMNUM  FOREIGN KEY(ITEMNUM )
REFERENCES CLOTHING(ITEMNUM );

ALTER TABLE CART
ADD CONSTRAINT FK_CART_MEMID  FOREIGN KEY(ID )
REFERENCES MEMBER(ID );	--아이디

CREATE SEQUENCE SEQ_CART
START WITH 1 INCREMENT BY 1;

CREATE TABLE INCOME(
SEQ NUMBER(20) NOT NULL,
ID VARCHAR2(50) NOT NULL, -- 주문자 아이디
ORDERNUM VARCHAR2(100) NOT NULL,  --주문번호
ITEMNUM NUMBER(8) NOT NULL, --상품 번호
TITLE VARCHAR2(50) NOT NULL,    --상품명
MAINPT VARCHAR2(50) NOT NULL,  --상품 썸네일 사진
PRICE NUMBER(20) NOT NULL,     --상품 가격
COUNT NUMBER(8) NOT NULL,
CANUM NUMBER(2)  NOT NULL,   --상품 카테고리 번호 (바지,긴판,반팔,아웃도어)
PDATE DATE NOT NULL      --결제일
);

CREATE SEQUENCE SEQ_INCOME
START WITH 1 INCREMENT BY 1;

CREATE TABLE ITEMREVIEW(
SEQ NUMBER(10) PRIMARY KEY, -- 시퀀스
ID VARCHAR2(50) NOT NULL, --작성자
ORDERNUM VARCHAR2(100) NOT NULL,  --주문번호(리뷰 작성하면 버튼 바꾸기위해)
ITEMNUM NUMBER(10), --어떤 상품 리뷰인가
TITLE VARCHAR2(1000) NOT NULL, -- 리뷰 제목
CONTENT VARCHAR2(1000) NOT NULL, --리뷰내용
RDATE DATE NOT NULL --작성날짜
);

CREATE SEQUENCE SEQ_ITEMREVIEW
START WITH 1
INCREMENT BY 1;

CREATE TABLE MEMBER(
SEQ NUMBER(5)  NOT NULL,
ID VARCHAR2(50) PRIMARY KEY,
PWD VARCHAR2(50) NOT NULL,
NAME VARCHAR2(100) NOT NULL,
NICKNAME  VARCHAR2(100)  UNIQUE NOT NULL,
GENDER VARCHAR2(20) NOT NULL,
PHONE VARCHAR2(50) NOT NULL,
EMAIL VARCHAR2(100) NOT NULL,
ADDRESS1 VARCHAR2(500) NOT NULL,
SNS_TYPE VARCHAR2(100),
AUTH NUMBER(1) NOT NULL,	-- 0일반회원 1 관리자 3 탈퇴
JOINDATE DATE NOT NULL --가입일
);

CREATE SEQUENCE SEQ_MEMBER
START WITH 1
INCREMENT BY 1;
